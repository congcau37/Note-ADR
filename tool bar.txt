 private void initToolbar() {
        toolbar = view.findViewById(R.id.toolbar);
        toolbar.setOnMenuItemClickListener(new Toolbar.OnMenuItemClickListener() {
            @Override
            public boolean onMenuItemClick(MenuItem item) {
                switch (item.getItemId()) {
                    case R.id.action_search_apply:
//                        String name = edtSearchName.getText().toString().trim();
//                        openAddingForm(new ProductResultSearch(), ConstantKey.KEY_PASS_PRODUCT_RESULT, filter(name, status, price));
                        searchServer();
                        break;
                    default:
                        return false;
                }
                return true;
            }
        });
        toolbar.inflateMenu(R.menu.menu_search_choice);
        toolbar.setNavigationIcon(R.drawable.ic_back);
        toolbar.setNavigationOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                getDialog().dismiss();
            }
        });
        toolbar.setTitle(getResources().getString(R.string.toolbar_title_search));
    }
