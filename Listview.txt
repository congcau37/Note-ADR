package demo.utt37.congcau.lisviewusingbaseadapter_kp.ui.adapter;

import android.content.Context;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.BaseAdapter;
import android.widget.ImageView;
import android.widget.TextView;

import java.util.ArrayList;
import java.util.List;
import java.util.zip.Inflater;

import demo.utt37.congcau.lisviewusingbaseadapter_kp.R;
import demo.utt37.congcau.lisviewusingbaseadapter_kp.model.CauThuModel;


public class CauThuAdapter extends BaseAdapter{

    private Context context;
    private int layout;
    private ArrayList<CauThuModel> cauThuModelList;

    public CauThuAdapter(Context context, int layout, ArrayList<CauThuModel> cauThuModelList) {
        this.context = context;
        this.layout = layout;
        this.cauThuModelList = cauThuModelList;
    }

    @Override
    public int getCount() {
        return cauThuModelList.size();
    }

    @Override
    public Object getItem(int i) {
        return null;
    }

    @Override
    public long getItemId(int i) {
        return 0;
    }

    @Override
    public View getView(int i, View view, ViewGroup viewGroup) {
        ViewHolder holder;
        if(view == null){
            holder = new ViewHolder();
            LayoutInflater inflater = (LayoutInflater) context.getSystemService(context.LAYOUT_INFLATER_SERVICE);
            view =inflater.inflate(layout,null);
            holder.txttenCauThu = (TextView) view.findViewById(R.id.txtTenCauThu);
            holder.txtquocTich = (TextView) view.findViewById(R.id.txtquocTich);
            holder.imgcauThu = (ImageView) view.findViewById(R.id.imgCauThu);
            holder.imgco = (ImageView) view.findViewById(R.id.imgco);
            view.setTag(holder);

        }else{

            holder= (ViewHolder) view.getTag();
        }

        CauThuModel cauThuModel = cauThuModelList.get(i);

        holder.txttenCauThu.setText(cauThuModel.getTenCauThu());
        holder.txtquocTich.setText(cauThuModel.getQuocTich());
        holder.imgcauThu.setImageResource(cauThuModel.getHinhCauThu());
        holder.imgco.setImageResource(cauThuModel.getHinhCo());

        return view;
    }
    public class ViewHolder{

        TextView txttenCauThu;
        TextView txtquocTich;
        ImageView imgcauThu;
        ImageView imgco;

    }

}
